{

\hspace{-6.5mm}\textcolor{Gray}{\rule[.75mm]{21.5mm}{1mm}} % make Gray line

}

	\item[
		{\makebox[1ex][r]{#1}}
		--
		{\makebox[1ex][l]{#2} }
		] }
